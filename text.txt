    <button id="spinBtn">SPIN</button>
    <p class="info">Every spin gives +8 coins 🎉</p>
  </div>

  <script>
    let coins = 0;
    let spinsLeft = 10;
    const symbols = ['🍒','🍀','⭐','💰','🎁','🔔'];

    const $ = id => document.getElementById(id);
    const updateHUD = () => {
      $('coins').textContent = coins;
      $('spinsLeft').textContent = spinsLeft;
    };

    function randSym(){ return symbols[Math.floor(Math.random()*symbols.length)]; }

    $('spinBtn').onclick = () => {
      if(spinsLeft<=0){
        alert('No spins left! Come back tomorrow.');
        return;
      }

      // Randomize display
      $('slot1').textContent = randSym();
      $('slot2').textContent = randSym();
      $('slot3').textContent = randSym();

      // Add fixed 8 coins
      coins += 8;
      spinsLeft--;

      updateHUD();
      setTimeout(() => alert('🎉 You won 8 coins!'), 50);
    };

    updateHUD();
  </script>
